 10:  * Translate a SQL query into it's frozen, vernacular form. This
 11:  * is how we get around differences in RDBMS query structure and
 12:  * syntax.
 13:  */
 14: class DialectTranslator {
 15:     const FALLBACK_DIALECT = "ANSI";
 16: 
 17:     protected $dialect;
 18:     protected $namespace;
 19: 
 20: 
 21:     /**
 22:      * Object constructor.
 23:      *
 24:      * @param string $dialect Which SQL dialect to translate into to
 25:      * @param string $namespace Namespace of dialect classes
 26:      */
 27:     public function __construct($dialect, $namespace = "Alchemy\\dialect") {
 28:         $this->dialect = $dialect;
 29:         $this->namespace = $namespace;
 30:     }
 31: 
 32: 
 33:     /**
 34:      * Get the name of the vernacular class which corresponds to the
 35:      * given class and dialect.
 36:      *
 37:      * @param string $master Class Name
 38:      * @param string $dialect
 39:      * @return string Vernacular Class Name
 40:      */
 41:     protected function getVernacularClassName($master, $dialect) {
 42:         // String namespace from class name
 43:         $cls = explode("\\", $master);
 44:         $cls = array_pop($cls);
 45: 
 46:         // Try Given dialect
 47:         $vern = "{$this->namespace}\\{$dialect}_{$cls}";
 48:         if (class_exists($vern)) {
 49:             return $vern;
 50:         }
 51: 
 52:         // Try ANSI
 53:         $vern = "{$this->namespace}\\" . self::FALLBACK_DIALECT . "_{$cls}";
 54:         if (class_exists($vern)) {
 55:             return $vern;
 56:         }
 57: 
 58:         // Traverse up inheritance tree
 59:         $parent = get_parent_class($master);
 60:         if ($parent) {
 61:             return $this->getVernacularClassName($parent, $dialect);
 62:         }
 63: 
 64:        // Die
 65:        throw new Exception("{$dialect} vernacular class does not exists for {$master}");
 66:     }
 67: 
 68: 
 69:     /**
 70:      * Translate a query into vernacular form
 71:      *
 72:      * @param IQuery $sqlexpr Query, optionally wrapped in a Monad
 73:      * @return ANSI_DialectBase
 74:      */
 75:     public function translate(IQuery $sqlexpr) {
 76:         return $this->translateImpl($sqlexpr);
 77:     }
 78: 
 79: 
 80:     /**
 81:      * Recursive implementation function for {@link DialectTranslator::translate()}
 82:      *
 83:      * @param mixed $sqlexpr
 84:      * @param array $settings
 85:      * @return ANSI_DialectBase
 86:      */
 87:     protected function translateImpl($sqlexpr, $settings = null) {
 88:         // Array?
 89:         if (is_array($sqlexpr)) {
 90:             $items = array();
 91:             foreach ($sqlexpr as $key => $item) {
 92:                 $items[$key] = $this->translateImpl($item, $settings);
 93:             }
 94:             return $items;
 95:         }
 96: 
 97:         // Scalar?
 98:         if (!is_object($sqlexpr)) {
 99:             return $sqlexpr;
100:         }
101: 
102:         // Find vernacular class
103:         $cls = get_class($sqlexpr);
104:         $cls = $this->getVernacularClassName($cls, $this->dialect);
105: 
106:         // Get Settings
107:         if (is_null($settings)) {
108:             $settings = $cls::settings();
109:         }
110: 
111:         // Convert object data into an array
112:         $data = array();
113:         foreach ((array)$sqlexpr as $key => $value) {
114:             if (is_object($value) || is_array($value)) {
115:                 $value = $this->translateImpl($value, $settings);
116:             }
117: 
118:             $key = trim($key, "\0* ");
119:             $data[$key] = $value;
120:         }
121: 
122:         // Create a vernacular version of the origin object
123:         $vernacular = new $cls($data, $settings);
124: 
125:         return $vernacular;
126:     }
127: }
