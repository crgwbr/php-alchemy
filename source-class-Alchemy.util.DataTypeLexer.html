 58:     /**
 59:      * Lexically analyze the given string into a stream of tokens.
 60:      *
 61:      * @param string $def
 62:      * @return array
 63:      */
 64:     protected function lexString($def) {
 65:         $def = str_split($def);
 66:         $tokens = array();
 67:         $inString = false;
 68:         $buffer = "";
 69: 
 70:         while (count($def) > 0) {
 71:             $char = array_shift($def);
 72: 
 73:             // Start / Stop String
 74:             if (preg_match("/[\"\']/", $char)) {
 75:                 $inString = !$inString;
 76:                 continue;
 77:             }
 78: 
 79:             // Escape character?
 80:             if ($char == "\\") {
 81:                 $buffer .= array_shift($def);
 82:                 continue;
 83:             }
 84: 
 85:             // Push value onto buffer
 86:             if ($inString || preg_match("/[a-zA-Z0-9_\-]/", $char)) {
 87:                 $buffer .= $char;
 88:                 continue;
 89:             }
 90: 
 91:             if (strlen($char) == 0 || $char === ' ') {
 92:                 continue;
 93:             }
 94: 
 95:             // Reached a control char. Record the token and reset the buffer
 96:             if (strlen($buffer) > 0) {
 97:                 $tokens[] = $this->normalizeToken($buffer);
 98:                 $buffer = "";
 99:             }
100: 
101:             $tokens[] = $char;
102:         }
103: 
104:         // Save last buffer
105:         if (strlen($buffer) > 0) {
106:             $tokens[] = $this->normalizeToken($buffer);
107:         }
108: 
109:         return $tokens;
110:     }

133:             // Float?

146:      * @param string $def

148:     protected function parse($def) {
149:         $tokens = $this->lexString($def);
150:         $this->type = array_shift($tokens);
151:         $this->args = array();
152:         $this->kwargs = array();
153: 
154:         while ($token = array_shift($tokens)) {
155:             // Ignore controls
156:             if (preg_match("/[\(\)\,]/", $token)) {
157:                 continue;
158:             }
159: 
160:             if (reset($tokens) == static::T_EQUALS) {
161:                 array_shift($tokens); // Swallow the assignment
162:                 $value = array_shift($tokens);
163:                 $this->kwargs[$token] = $value;
164:             } else {

166:             }
167:         }
168:     }
