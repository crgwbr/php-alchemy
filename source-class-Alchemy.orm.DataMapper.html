  9:  * Abstract base class for custom ORM models. this class
 10:  * doesn't actually store any data, it merely acts as an
 11:  * interface between Domain objects and the Session (which
 12:  * actually owns all of the DB records).
 13:  */

 19:      */

 22:     protected static $indexes = array();

 34:      * @param mixed $sessionID A pointer to the record for this objects data in the session
 35:      */
 36:     public static function from_session(Session $session, $sessionID) {
 37:         $cls = get_called_class();

 45:      * Gen an instance of Table that represents the schema of this
 46:      * domain object.
 47:      *
 48:      * @return Table
 49:      */

 51:         $cls = get_called_class();
 52:         if (!array_key_exists($cls, self::$schema_cache)) {
 53:             $table = new Table($cls::table_name(), $cls::$props, $cls::$indexes);
 54:             self::$schema_cache[$cls] = $table;
 55:         }
 56: 
 57:         return self::$schema_cache[$cls];
 58:     }

 61:     /**
 62:      * Get the table name for this mapper
 63:      *
 64:      * @return string Table Name
 65:      */
 66:     public static function table_name() {
 67:         $cls = static::$table_name ?: get_called_class();
 68:         $cls = preg_replace("/[^A-Za-z0-9]/", "_", $cls);
 69:         return $cls;
 70:     }

 73:     /**
 74:      * Get the value of a column on this object
 75:      *
 76:      * @param string $prop
 77:      * @return mixed
 78:      */
 79:     public function __get($prop) {
 80:         $table = static::table();
 81:         if (!$table->isColumn($prop)) {
 82:             throw new Exception("Property [{$prop}] is not a configured column");
 83:         }
 84: 
 85:         $cls = get_called_class();
 86:         if (array_key_exists($prop, $this->deltas)) {
 87:             return $this->deltas[$prop];
 88:         }
 89: 
 90:         if (is_object($this->session)) {
 91:             return $this->session->getProperty($cls, $this->sessionID, $prop);
 92:         }
 93: 
 94:         return null;
 95:     }

108:             throw new Exception("Property [{$prop}] is not a configured column");

141:     /**
142:      * Save this object's data back to the session. By default this will queue
143:      * an UPDATE statement to be run on the server as soon as {@Session::commit()}
144:      * is called.
145:      *
146:      * @param bool $queueUpdate Default's to true
147:      */
148:     public function save($queueUpdate = true) {
149:         if (!$this->session) {
150:             throw new Exception("Can not save this DataMapper because it is not associated with a session");
151:         }
152: 
153:         $cls = get_class($this);
154:         foreach ($this->deltas as $prop => $value) {
155:             $this->session->setProperty($cls, $this->sessionID, $prop, $value);
156:         }
157: 
158:         if ($queueUpdate) {
159:             $this->session->save($cls, $this->sessionID);
160:         }
161: 
162:         $this->deltas = array();
163:     }

171:      */
172:     public function setSession(Session $session, $sessionID) {

174:         $this->sessionID = $sessionID;
